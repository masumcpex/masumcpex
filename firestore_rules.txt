rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // kh_members এবং kh_records — যে কেউ পড়তে ও লিখতে পারবে, কোনো লগইন লাগবে না।
    match /kh_members/{memberId} {
      allow read, write: if true;
    }
    match /kh_records/{recordId} {
      allow read, write: if true;
    }

    // বাকি সব কিছুর জন্য ডিফল্টভাবে বন্ধ (নিরাপত্তার জন্য)
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
